text to image - Graphics 2D API

public class Text_to_image
{
    public static void main(String[] args) throws IOException
    {
        String text = "John walked";
        int text_size = 65;
        int x = 500;
        int y = 80;        

        BufferedImage img = new BufferedImage
        (x, y, BufferedImage.TYPE_INT_RGB);
        
        Graphics2D g2d = img.createGraphics();
        
        g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, 
                RenderingHints.VALUE_ANTIALIAS_ON);
        
        Font font = new Font("Arial", Font.PLAIN, text_size);
        g2d.setFont(font);
        
        g2d.setColor(Color.WHITE);
        g2d.fillRect(0, 0, x, y);
        
        g2d.setColor(Color.BLACK);
        g2d.drawString(text, 0, text_size);
        
        ImageIO.write(img, "png", new File("file.png"));
    }
}
