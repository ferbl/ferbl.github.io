<!DOCTYPE html>
<html>
    
    <head>   
        <script>
            function start_recognition()
            {
                var rec = new webkitSpeechRecognition();
            
                rec.lang = "en-US";
                rec.start();
                
                rec.onresult = function(e)
                {
                    document.write(e.results[0][0].transcript);
                    rec.stop();
                };
            }
        </script>
    </head>

    <body>

        <p id="output" onclick="start_recognition()">...</p>

    </body>

</html>
